% Adjusted corner command
\newcommand{\corner}[1]{%
  \begin{tikzpicture}[remember picture, overlay]
    \node[anchor=north west] at ([xshift=\borderoffset, yshift=-\borderoffset]current page.north west){%
      \pgfornament[width=2cm, color=victorianred]{#1}};
    \node[anchor=north east] at ([xshift=-\borderoffset, yshift=-\borderoffset]current page.north east){%
      \pgfornament[width=2cm, symmetry=v, color=victorianred]{#1}};
    \node[anchor=south west] at ([xshift=\borderoffset, yshift=\borderoffset]current page.south west){%
      \pgfornament[width=2cm, symmetry=h, color=victorianred]{#1}};
    \node[anchor=south east] at ([xshift=-\borderoffset, yshift=\borderoffset]current page.south east){%
      \pgfornament[width=2cm, symmetry=c, color=victorianred]{#1}};
  \end{tikzpicture}%
}
