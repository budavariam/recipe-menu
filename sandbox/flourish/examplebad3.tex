% Adjusted complete frame with flourishes connecting corners
\newcommand{\victorianframeadjusted}[2]{%
  \begin{tikzpicture}[remember picture, overlay]
    % Corner ornaments
    \node[anchor=north west] at ([xshift=2cm, yshift=-2cm]current page.north west){%
      \pgfornament[width=2cm, color=victorianred]{#1}};
    \node[anchor=north east] at ([xshift=-2cm, yshift=-2cm]current page.north east){%
      \pgfornament[width=2cm, symmetry=v, color=victorianred]{#1}};
    \node[anchor=south west] at ([xshift=2cm, yshift=2cm]current page.south west){%
      \pgfornament[width=2cm, symmetry=h, color=victorianred]{#1}};
    \node[anchor=south east] at ([xshift=-2cm, yshift=2cm]current page.south east){%
      \pgfornament[width=2cm, symmetry=c, color=victorianred]{#1}};
    
    % Top and bottom flourishes
    \node[anchor=north] at ([yshift=-2cm]current page.north){%
      \pgfornament[width=6.5cm, symmetry=h, color=victorianred]{#2}};
    \node[anchor=south] at ([yshift=2cm]current page.south){%
      \pgfornament[width=6.5cm, color=victorianred]{#2}};
      
    % Left and right connecting flourishes (rotated)
    \node[anchor=west, rotate=90] at ([xshift=2cm]current page.west){%
      \pgfornament[width=6.5cm, symmetry=h, color=victorianred]{#2}};
    \node[anchor=east, rotate=-90] at ([xshift=-2cm]current page.east){%
      \pgfornament[width=6.5cm, color=victorianred]{#2}};
      
    % Draw connecting lines between ornaments - top and bottom
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(0cm,0.1cm)}]current page.north west) -- 
      ([shift={(-8.7cm,0.1cm)}]current page.north);
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(8.7cm,0.1cm)}]current page.north) -- 
      ([shift={(-0cm,0.1cm)}]current page.north east);
      
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(0cm,-0.1cm)}]current page.south west) -- 
      ([shift={(-8.7cm,-0.1cm)}]current page.south);
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(8.7cm,-0.1cm)}]current page.south) -- 
      ([shift={(-0cm,-0.1cm)}]current page.south east);
      
    % Draw connecting lines between ornaments - left and right
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(0.1cm,-2cm)}]current page.north west) -- 
      ([shift={(0.1cm,8.5cm)}]current page.west);
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(0.1cm,-8.5cm)}]current page.west) -- 
      ([shift={(0.1cm,2cm)}]current page.south west);
      
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(-0.1cm,-2cm)}]current page.north east) -- 
      ([shift={(-0.1cm,8.5cm)}]current page.east);
    \draw[line width=0.5pt, color=victorianred] 
      ([shift={(-0.1cm,-8.5cm)}]current page.east) -- 
      ([shift={(-0.1cm,2cm)}]current page.south east);
  \end{tikzpicture}%
}
